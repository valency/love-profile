1 /**
  2  * @namespace 处理浏览器和flash插件识别
  3  * @private
  4  */
  5 var Browser = {
  6 	getBrowser : function() {
  7 		if (!Browser.browser) {
  8 			var ua = navigator.userAgent.toLowerCase(), platform = navigator.platform
  9 					.toLowerCase(), UA = ua
 10 					.match(/(opera|ie|firefox|chrome|version)[\s\/:]([\w\d\.]+)?.*?(safari|version[\s\/:]([\w\d\.]+)|$)/)
 11 					|| [null, 'unknown', 0], mode = UA[1] == 'ie'
 12 					&& document.documentMode;
 13 
 14 			Browser.browser = {
 15 				name : (UA[1] == 'version') ? UA[3] : UA[1],
 16 
 17 				version : mode
 18 						|| parseFloat((UA[1] == 'opera' && UA[4])
 19 								? UA[4]
 20 								: UA[2]),
 21 
 22 				platform : ua.match(/ip(?:ad|od|hone)/) ? 'ios' : (ua
 23 						.match(/(?:webos|android)/)
 24 						|| platform.match(/mac|win|linux/) || ['other'])[0]
 25 			}
 26 		}
 27 		return Browser.browser;
 28 	},
 29 
 30 	getFlash : function() {
 31 		if (!Browser.flash) {
 32 			var version = '0 r0';
 33 			try {
 34 				version = navigator.plugins['Shockwave Flash'].description;
 35 			} catch (e) {
 36 				try {
 37 					version = new ActiveXObject('ShockwaveFlash.ShockwaveFlash')
 38 							.GetVariable('$version');
 39 				} catch (ex) {
 40 				}
 41 			}
 42 			version = version.match(/\d+/g);
 43 			Browser.flash = {
 44 				version : Number(version[0] || '0.' + version[1]) || 0,
 45 				build : Number(version[2]) || 0
 46 			};
 47 		}
 48 		return Browser.flash;
 49 	}
 50 };
