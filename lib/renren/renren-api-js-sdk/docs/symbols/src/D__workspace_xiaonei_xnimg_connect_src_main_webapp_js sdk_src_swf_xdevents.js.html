  3 	Sender : new Class({

  6

  8

 12 			addReceiver.internal = removeReceiver.internal = true;
 13 			this.parent({
 14 						addReceiver : addReceiver,
 15 						removeReceiver : removeReceiver
 16 					}, acceptOrigins);

 18

 20 			if (msg.origin && !this.$receivers[origin])
 21 				this.$receivers[origin] = {
 22 					transport : msg.transport,
 23 					origin : msg.origin,
 24 					source : msg.source
 25 				};
 26 		},
 27 
 28 		removeReceiver : function(msg) {

 30 				delete this.$receivers[origin];

 34 			args = args || [];
 35 			var t = typeOf(args);
 36 			if (t != 'array' && t != 'arguments')
 37 				args = [args];
 38 			if (delay != null)
 39 				extend(args, delay);
 40 			forEach(this.$receivers, function(receiver) {
 41 						XD.sendMessage({
 42 									type : type,
 43 									args : args,
 44 									delay : delay
 45 								}, receiver.origin, receiver.transport,
 46 								receiver.source);
 47 					}, this);
 48 			return this;
 49 		}
 50 	}),
 51 
 52 	Receiver : new Class({
 53 
 54 				Extends : XDPC.Client,
 55 
 56 				Implements : [Events],
 57

 60 					this.call('addReceiver');

 62

 64 					if (msg.data && msg.data.type) {
 65 						if (msg.data.type && this.checkOrigin(msg.origin)) {
 66 							this.fireEvent(msg.data.type, msg.data.args,
 67 									msg.data.delay);
 68 						}
 69 					}
 70 				},
 71 
 72 				addEvent : function(type, fn, internal) {
 73 					type = removeOn(type);
 74 					this.$events[type] = (this.$events[type] || []);
 75 					if (indexOf(this.$events[type], fn) == -1)
 76 						this.$events[type].push(fn);
 77 					if (internal)
 78 						fn.internal = true;
 79 					return this;
 80 				}
 81 			})
