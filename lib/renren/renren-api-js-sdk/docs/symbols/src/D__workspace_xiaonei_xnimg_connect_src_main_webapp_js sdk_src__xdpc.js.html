1 var XDPC = {
  2 	Server : new Class(
  3 			/**
  4 			 * @lends Renren.XDPC.Server.prototype
  5 			 */
  6 			{
  7 		methods : {},
  8 
  9 		/**
 10 		 * @class 模似rpc实现在异域窗口间方法的调用
 11 		 * @constructs
 12 		 * @param methods
 13 		 *            {Array} 要公开可被跨域调用的方法
 14 		 * @param clientOrigins
 15 		 *            {String | Array} (optional)
 16 		 *            允许调用的页面来源，如允许来自'http://www.example.com/'下的页面的调用
 17 		 *            @example
 18 		 *            new Renren.XDPC.Server({'size': setSize, 'trace': trace}, ['http://www.example.com/']);
 19 		 */
 20 		initialize : function(methods, clientOrigins) {
 21 			XD.addEvent('receivedXDMessage', bind(this.handleXDMessages, this));
 22 			if (methods) {
 23 				this.addMethods(methods, clientOrigins);
 24 			}
 25 		},
 26 
 27 		/**
 28 		 * 公开单个跨域方法
 29 		 * 
 30 		 * @param name
 31 		 *            {String} 公开的跨域方法名
 32 		 * @param fn
 33 		 *            {Function} 公开的跨域方法
 34 		 * @param orgns
 35 		 *            {Array} (optional) 允许调用的页面来源
 36 		 */
 37 		addMethod : function(name, fn, orgns) {
 38 			if (!this.methods[name]) {
 39 				this.methods[name] = XDPC.proxy(name, fn, orgns
 40 								&& typeOf(orgns) != 'array' ? [orgns] : orgns);
 41 			}
 42 		},
 43 
 44 		/**
 45 		 * 公开多个跨域方法
 46 		 * 
 47 		 * @param name
 48 		 *            {Object} 公开的多个跨域方法，方法名为key，方法体为value
 49 		 * @param clientOrgns
 50 		 *            (optional) {String | Array} 允许调用的页面来源
 51 		 */
 52 		addMethods : function(methods, clientOrgns) {
 53 			if (clientOrgns && typeOf(clientOrgns) != 'array') {
 54 				clientOrgns = [clientOrgns];
 55 			}
 56 			var orgns = append([], clientOrgns);
 57 			for (var name in methods) {
 58 				this.addMethod(name, methods[name], orgns);
 59 			}
 60 		},
 61 
 62 		/**
 63 		 * 移除一个跨域方法
 64 		 * 
 65 		 * @param name
 66 		 *            {String} 跨域方法名
 67 		 */
 68 		removeMethod : function(name) {
 69 			if (this.methods[name])
 70 				delete this.methods[name];
 71 		},
 72 
 73 		/**
 74 		 * @private
 75 		 */
 76 		handleXDMessages : function(msg) {
 77 			if (msg.data && msg.data.method) {
 78 				var proxyMethod = this.methods[msg.data.method];
 79 				if (proxyMethod) {
 80 					proxyMethod(msg);
 81 				}
 82 			}
 83 		}
 84 	}),
 85 
 86 	proxy : function(name, fn, origins) {
 87 		return function(msg) {
 88 			if (origins && origins.length > 0
 89 					&& !XDPC.checkOrigin(msg.origin, origins))
 90 				return;
 91 			var r = fn.apply(null, fn.internal ? msg : msg.data.args);
 92 			if (msg.data.cbid)
 93 				XD.sendMessage({
 94 							result : r,
 95 							cbid : msg.data.cbid
 96 						}, msg.origin, msg.transport, msg.source);
 97 		}
 98 	},
 99 
100 	checkOrigin : function(orgn, origins) {
101 		if (!orgn)
102 			return false;
103 		for (var i = 0, l = origins.length; i < l; i++) {
104 			if (origins[i].indexOf(orgn) == 0 || orgn.indexOf(origins[i]) == 0)
105 				return true;
106 		}
107 		return false;
108 	},
109 
110 	Client : new Class(
111 			/**
112 			 * @lends Renren.XDPC.Client.prototype
113 			 */
114 			{
115 
116 		callbacks : {},
117 
118 		/**
119 		 * @constructs
120 		 * @param server
121 		 *            {Object} 要调用的提供方法的窗口的跨域信息
122 		 *            @example
123 		 *            new Renren.XDPC.Client({relation:'parent', origin: 'http://apps.renren.com/'});
124 		 */
125 		initialize : function(server) {
126 			XD.addEvent('receivedXDMessage', bind(this.handleXDMessages, this));
127 			this.server = merge({}, server);
128 		},
129 
130 		/**
131 	 * @private
132 	 */
133 		handleXDMessages : function(msg) {
134 			if (msg.data && msg.data.cbid) {
135 				var cb = this.callbacks[msg.data.cbid];
136 				if (cb && this.checkOrigin(msg.origin)) {
137 					delete this.callbacks[msg.data.cbid];
138 					cb(msg.data.result);
139 				}
140 			}
141 		},
142 
143 		/**
144 		 * @private
145 		 */
146 		checkOrigin : function(origin) {
147 			return XDPC.checkOrigin(origin, [this.server.origin])
148 		},
149 
150 		/**
151 		 * @param name
152 		 *            {String} 跨域方法名
153 		 * @param args
154 		 *            {Array} 跨域方法参数
155 		 * @param callback
156 		 *            {Function} (optional) 跨域方法执行完后再跨域回调，为空时将不进行回调
157 		 */
158 		call : function(name, args, callback) {
159 			var ts = XD.transports, trans = ts.postMessage || ts.flash, tp = typeOf(args), cbid;
160 			if (!trans)
161 				return;
162 			if (callback) {
163 				cbid = guid();
164 				this.callbacks[cbid] = callback;
165 			}
166 			if (tp != 'array' && tp != 'arguments')
167 				args = [args];
168 			XD.sendMessage({
169 						method : name,
170 						args : args,
171 						cbid : cbid
172 					}, this.server.origin, trans, this.server.relation);
173 		}
174 	})
175 };
