1 /**
  2  * @namespace 对进行跨域通信用的flash swf文件的封装
  3  * @private
  4  */
  5 var Flash = {
  6 	url : rrUrls.widget + 'XdComm.swf',
  7 
  8 	name : 'RenrenXdSwf',
  9 
 10 	callbacks : [],
 11 
 12 	init : function() {
 13 		if (Flash.inited) {
 14 			return;
 15 		}
 16 		Flash.inited = true;
 17 		var html = ('<object type="application/x-shockwave-flash" id="'
 18 				+ Flash.name
 19 				+ '" data="'
 20 				+ Flash.url
 21 				+ '"'
 22 				+ (!!document.attachEvent
 23 						? ' name="'
 24 								+ Flash.name
 25 								+ '" classid="clsid:d27cdb6e-ae6d-11cf-96b8-444553540000"'
 26 						: '')
 27 				+ ' width="0" height="0"><param name="movie" value="'
 28 				+ Flash.url + '"></param><param name="allowScriptAccess" value="always"></param><param name="flashVars" value="onReady=Renren.onFlashReady"/></object>');
 29 		Dom.appendHidden(html);
 30 	},
 31 
 32 	onFlashReady : function() {
 33 		Flash.XdComm = document[Flash.name];
 34 		for (var i = 0, l = Flash.callbacks.length; i < l; i++) {
 35 			Flash.callbacks[i]();
 36 		}
 37 		Flash.callbacks = [];
 38 	},
 39 
 40 	hasMinVersion : function() {
 41 		return Browser.getFlash().version >= 9;
 42 	},
 43 
 44 	onReady : function(cb) {
 45 		Flash.init();
 46 		if (Flash.XdComm) {
 47 			window.setTimeout(function() {
 48 						cb()
 49 					}, 0);
 50 		} else {
 51 			Flash.callbacks.push(cb);
 52 		}
 53 	},
 54 
 55 	initPostMessage : function(cb, origin) {
 56 		return Flash.XdComm.postMessage_init(cb, origin);
 57 	},
 58 
 59 	postMessage : function(message, targetOrigin) {
 60 		return Flash.XdComm.postMessage_send(message, targetOrigin);
 61 	},
 62 	
 63 	sendXdHttpRequest : function(method, url, params, headers, callback, reqId) {
 64 		return Flash.XdComm.sendXdHttpRequest(method, url, params, headers, callback, reqId);
 65 	}
 66 }
