1 /**
  2  * 用以弹出各种对话界面，如发送新鲜事，授权，邀请。
  3  * @methodOf Renren
  4  * @param options {Object}
  5  * @param options.url {String}
  6  *            请求dialog url或oauth
  7  *            url，非http、https开头的值将视为相对于http://widget.renren.com/dialog/的路径
  8  * @param options.method {String} (optional)
  9  *            http请求方法，可选值：post、get
 10  * @param options.show {String}
 11  *            页面在何处显示，可选值：popup（新窗口）、iframe（类似dialog的效果）、page（当前窗口）           
 12  * @param options.style {Object} (optional)
 13  *            ui的样式如：{top : 100, left : 300, width : 450, height :
 14  *            350}，相应值没设置时会根据窗口大小在合适位置显示
 15  * @param options.params {Object}
 16  *            请求所需参数值，因调用dialog而不同，相关所需参数可见：<a href="http://www.example.com">Example Project</a>.
 17  * @param options.fallback {Boolean} (optional)
 18  *            默认为true，做些容错处理，如当请求参数过长超过浏览器限制，将转化http请求方法为post;若用popup显示时被浏览器
 19  *            阻止弹窗时将用当前窗口显示，完成后再跳转回当前页，即同options.show = page。若需自己处理这些情况时可设值为false。
 20  * @param options.onSuccess {Function} (optional)
 21  *            请求成功后的回调函数，请求结果做为回调函数的参数传入
 22  * @param options.onFailure {Function} (optional)
 23  *            请求失败后的回调函数，请求结果做为回调函数的参数传入
 24  * @param options.onComplete {Function} (optional)
 25  *            请求完成后的回调函数，不论请求成功，失败都会触发
 26  * @param cb {Function} (optional)
 27  *            请求完成后的回调函数作用同options.onComplete，但options.onComplete有值则此值不起作用
 28  * @example
 29  * Renren.ui({
 30  *	url : 'feed',
 31  *	show : 'popup',
 32  *	style : {
 33  *		top : 50
 34  *	},
 35  *	params : {
 36  *		url:'http://www.swimmingacross.com',
 37  *		name: '新鲜事测试',
 38  *		description:'测试',
 39  *		image:'http://at-img4.tdimg.com/board/2011/5/46465.jpg'
 40  *	},
 41  *	onComplete: function(r){console.log("complete:" + toJSON(r));},
 42  *	onFailure: function(r){console.log("failure:" + r.error + ' ' + r.error_description);} 
 43  * });           
 44  */
 45 var ui = function(options, cb) {
 46 	if (!options.url) {
 47 		throw new Error('The url argument must not be null.');
 48 	}
 49 	var url = options.url, url = (url.indexOf('http://') == 0 || url
 50 			.indexOf('https://') == 0)
 51 			? url
 52 			: (rrUrls.widget + 'dialog/' + url);
 53 	var p = /^(https?:\/\/[^\/]*\/)([^\/\?#]*\/)*([^\/\?#]*)/.exec(url);
 54 	if (p == null
 55 			|| (rrUrls.widget != p[1] && rrUrls.graph != p[1] && rrUrls.graph_https != p[1])) {
 56 		return;
 57 	}
 58 	var reqOptns = combine({
 59 				url : url,
 60 				method : (options.method
 61 						&& options.method.toLowerCase() === 'post'
 62 						? 'post'
 63 						: 'get')
 64 			}, options, defaults[(p[3] || '').toLowerCase()], {
 65 				show : 'popup',
 66 				style : {
 67 					width : 475,
 68 					height : 340
 69 				},
 70 				params : {
 71 					app_id : rrUrls.widget == p[1]
 72 							? Renren.options.appId
 73 							: null,
 74 					client_id : (rrUrls.graph == p[1] || rrUrls.graph_https == p[1])
 75 							? Renren.options.appId
 76 							: null
 77 				},
 78 				onComplete : cb
 79 			});
 80 
 81 	var showIn = reqOptns.show, request;
 82 	if (showIn.length > 0) {
 83 		showIn = showIn.substr(0, 1).toUpperCase() + showIn.substr(1);
 84 	}
 85 	if (Request[showIn]) {
 86 		request = new Request[showIn](reqOptns);
 87 	}
 88 	if (!request) {
 89 		throw new Error("Fail to start an ui request, the show argument may be invalid.");
 90 	}
 91 	return request.send();
 92 };
 93 
 94 var defaults = {
 95 	'authorize' : {
 96 		show : 'page',
 97 		style : {
 98 			width : 570,
 99 			height : 340
100 		}
101 	}
102 };
