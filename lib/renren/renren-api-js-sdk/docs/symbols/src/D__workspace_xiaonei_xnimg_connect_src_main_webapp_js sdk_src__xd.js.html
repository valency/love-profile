1 /**
  2  * 
  3  * @namespace 跨域通信
  4  * @memberOf Renren
  5  * @see <a
  6  *      href="https://developer.mozilla.org/en/DOM/window.postMessage">window.postMessage</a>.
  7  * @see <a
  8  *      href="http://livedocs.adobe.com/flash/9.0/ActionScriptLangRefV3/flash/net/LocalConnection.html">flash
  9  *      LocalConnection</a>.
 10  */
 11 var XD =
 12 /**
 13  * @lends Renren.XD
 14  */
 15 {
 16 
 17 	/**
 18 	 * {String} 在跨域通信中标识当前页面，可参见<a
 19 	 * href="https://developer.mozilla.org/en/DOM/window.postMessage">window.postMessage</a>中所定义的origin
 20 	 */
 21 	origin : (location.protocol + '//' + location.host + '/' + guid()),
 22 
 23 	/**
 24 	 * {Object} 浏览器支持的跨域通信方式
 25 	 */
 26 	transports : {
 27 		fragment : 'fragment',
 28 		postMessage : window.postMessage ? undefined : false
 29 	},
 30 
 31 	/**
 32 	 * 初始化方法，在Renren.init中被调用，根椐浏览器支持情况选择跨域通信方式
 33 	 * 
 34 	 * @param options
 35 	 *            {Object}
 36 	 * @param options.originPath
 37 	 *            {String} (optional) 用于构成跨域的origin值，origin用以唯一标识当前本次加载的页面
 38 	 */
 39 	init : function(options) {
 40 		if (options && options.originPath != undefined)
 41 			XD.origin = location.protocol + '//' + location.host + '/'
 42 					+ options.originPath;
 43 		if (XD.transports.postMessage == undefined) {
 44 			var bws = Browser.getBrowser();
 45 			// window.opener can't work in IE8
 46 			if (bws.name == 'ie' && bws.version == 8)
 47 				XD.transports.postMessage = false;
 48 			else
 49 				XD.PostMessage.init();
 50 		}
 51 		if (!XD.transports.postMessage && XD.transports.flash == undefined) {
 52 			XD.transports.flash = Flash.hasMinVersion() ? 0 : false;
 53 			XD.transports.flash === 0 && XD.Flash.init();
 54 		}
 55 	},
 56 
 57 	/**
 58 	 * @private
 59 	 */
 60 	resolveRelation : function(relation) {
 61 		var pt, matches, parts = relation.split('.'), node = window;
 62 		for (var i = 0, l = parts.length; i < l; i++) {
 63 			pt = parts[i];
 64 			if (pt == 'opener' || pt == 'parent' || pt == 'top') {
 65 				node = node[pt];
 66 			} else if (matches = /^frames\[['"]?([a-zA-Z0-9-_]+)['"]?\]$/
 67 					.exec(pt)) {
 68 				node = node.frames[matches[1]];
 69 			} else {
 70 				throw new SyntaxError('Malformed value to resolve: ' + relation);
 71 			}
 72 		}
 73 		return node;
 74 	},
 75 
 76 	/**
 77 	 * @namespace 使用window.postMessage实现跨域通信
 78 	 * @private
 79 	 */
 80 	PostMessage : {
 81 		init : function() {
 82 			window.addEventListener ? window.addEventListener('message',
 83 					XD.PostMessage.receiveMessage, false) : window.attachEvent(
 84 					'onmessage', XD.PostMessage.receiveMessage);
 85 			XD.transports.postMessage = 'postMessage';
 86 		},
 87 
 88 		receiveMessage : function(event) {
 89 			XD.receiveMessage({
 90 						data : event.data,
 91 						origin : event.origin,
 92 						source : event.source,
 93 						transport : 'postMessage'
 94 					});
 95 		},
 96 
 97 		sendMessage : function(message, targetOrigin, targetWin) {
 98 			if (typeOf(message) != 'string')
 99 				message = toQueryString(message);
100 			targetWin.postMessage(message, targetOrigin);
101 		}
102 	},
103 
104 	/**
105 	 * @namespace 使用flash实现跨域通
106 	 * @private
107 	 */
108 	Flash : {
109 		init : function() {
110 			Flash.onReady(function() {
111 						XD.transports.flash = Flash.initPostMessage(
112 								'Renren.XD.Flash.receiveMessage', XD.origin)
113 								? 'flash'
114 								: false;
115 					});
116 		},
117 
118 		receiveMessage : function(message) {
119 			XD.receiveMessage(message);
120 		},
121 
122 		sendMessage : function(message, targetOrigin) {
123 			return Flash.postMessage(message, targetOrigin);
124 		}
125 	},
126 
127 	/**
128 	 * @namespace 用同域页面实现跨域通信
129 	 * @private
130 	 */
131 	Fragment : {
132 		magic : 'rr_xd_fragment',
133 
134 		sendMessage : function(message, targetOrigin, targetWin) {
135 			if (targetOrigin
136 					&& targetOrigin.indexOf(targetWin.location.protocol + '//'
137 							+ targetWin.location.host + '/') == 0)
138 				targetWin.Renren.XD.receiveMessage({
139 							data : message
140 						});
141 		}
142 	},
143 
144 	/**
145 	 * @private
146 	 */
147 	receiveMessage : function(msg) {
148 		if (typeOf(msg.data) == 'string') {
149 			msg.data = fromQueryString(msg.data);
150 		}
151 		XD.fireEvent("receivedXDMessage", msg);
152 	},
153 
154 	/**
155 	 * @param message
156 	 *            {Object | String } 发送信息
157 	 * @param targetOrigin
158 	 *            {String} 接收窗体的页面标识
159 	 * @param transport
160 	 *            {String} 跨域通信方式
161 	 * @param targetRelWin
162 	 *            {String | Window} 接收窗体的window对象或和本窗体的关系
163 	 */
164 	sendMessage : function(message, targetOrigin, transport, targetRelWin) {
165 		if (targetOrigin && transport && transport.length > 0) {
166 			transport = transport.substr(0, 1).toUpperCase()
167 					+ transport.substr(1);
168 			try {
169 				if (typeof targetRelWin == 'string') {
170 					targetRelWin = XD.resolveRelation(targetRelWin);
171 				}
172 				XD[transport]
173 						&& XD[transport].sendMessage(message, targetOrigin,
174 								targetRelWin);
175 			} catch (e) {
176 			}
177 		}
178 	},
179 
180 	/**
181 	 * @private
182 	 */
183 	dispatchLocationFragment : function(loc) {
184 		loc = loc || location.toString();
185 		if (loc.indexOf('#') == -1)
186 			return;
187 		var fragment = loc.substr(loc.indexOf('#') + 1), magicIndex = fragment
188 				.indexOf(XD.Fragment.magic), params = fromQueryString(fragment);
189 		if (magicIndex > 0) {
190 			document.documentElement.style.display = 'none';
191 		}
192 		params.origin
193 				&& XD.sendMessage(params, params.origin, params.transport,
194 						params.relation);
195 	}
196 };
197 
198 XD.dispatchLocationFragment();
