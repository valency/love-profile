1 Request.Flash = new Class({
  2 	Extends : Request,
  3 
  4 	options : {
  5 		fallback : true,
  6 		params : {
  7 			format : 'JSON'
  8 		}
  9 	},
 10 
 11 	initialize : function(options) {
 12 		this.parent(options);
 13 		var self = this;
 14 		this.addEvents({
 15 					'request' : function() {
 16 						Request.Flash.addRequest(self);
 17 					},
 18 					'complete' : function() {
 19 						Request.Flash.removeRequest(self);
 20 					}
 21 				});
 22 	},
 23 
 24 	send : function() {
 25 		if (!this.check())
 26 			return this;
 27 		this.running = true;
 28 		var opts = this.options, method = opts.method, url = opts.url, params = opts.params, qs = toQueryString(opts.params), self = this;
 29 		if (method === 'get') {
 30 			url += (opts.url.indexOf('?') < 0 ? '?' : '&') + qs;
 31 			qs = '';
 32 		} else if (method !== 'post') {
 33 			method = 'post';
 34 			qs = toQueryString(combine(params, {
 35 						_method : method
 36 					}));
 37 		}
 38 		Flash.onReady(function() {
 39 			self.fireEvent('request');
 40 			var r = Flash.sendXdHttpRequest(method.toUpperCase(), url, qs,
 41 					null, 'Renren.Request.Flash.onResponse', self.id);
 42 			if (r == null)
 43 				self.abort({
 44 							error : 'abort',
 45 							error_description : 'Send flash XdHttpRequest error.'
 46 						});
 47 		});
 48 		return this;
 49 	},
 50 
 51 	handleResponse : function(data) {
 52 		this.response = data;
 53 		this.running = false;
 54 		if (typeOf(data) == 'string' && validateJSON(data)) {
 55 			data = parseJSON(data);
 56 		}
 57 		if (!data.error_code)
 58 			this.success(this.response);
 59 		else
 60 			this.failure(this.response);
 61 	}
 62 });
 63 
 64 extend(Request.Flash, {
 65 			requests : {},
 66 
 67 			onResponse : function(reqId, data) {
 68 				if (reqId != null) {
 69 					var req = this.requests[reqId];
 70 					if (req) {
 71 						req.handleResponse(data);
 72 					}
 73 				}
 74 			},
 75 
 76 			addRequest : function(req) {
 77 				if (!this.requests[req.id])
 78 					this.requests[req.id] = req;
 79 			},
 80 
 81 			removeRequest : function(req) {
 82 				if (this.requests[req.id])
 83 					delete this.requests[req.id];
 84 			}
 85 		});
